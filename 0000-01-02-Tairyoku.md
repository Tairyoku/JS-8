function finalGrade(examGrade, numOfProjects) {
  if (examGrade > 90 || numOfProjects > 10 ) return 100; 
  else if (examGrade > 75 || numOfProjects > 5 ) return 90; 
  else if (examGrade > 50 || numOfProjects > 2 ) return 75;
  else return 0;
}
//Проверка
alert(finalGrade(100, 12));
alert(finalGrade(99, 0));
alert(finalGrade(10, 15));
alert(finalGrade(85, 5));
alert(finalGrade(55, 3));
alert(finalGrade(55, 0));
alert(finalGrade(20, 2));
